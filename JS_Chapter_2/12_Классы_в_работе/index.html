Создаем карточки динамически с помощтю классов

class menuCard {
constructor(img, alt, subtitle, descr, price, parentSelector) {
  this.img = img;
  this.alt = alt;
  this.subtitle = subtitle;
  this.descr = descr;
  this.price = price;
  this.parent = document.querySelector(parentSelector);
  this.transfer = 27;
  this.currencyConverter();
}

currencyConverter() {
  this.price = +this.price * this.transfer;
}

render() {
  const element = document.createElement("div");
  element.innerHTML = `
        

            
            
${this.subtitle}

            
${this.descr}

            

            

                
Цена:

                
${this.price} грн/день

            

        

    `;
  this.parent.append(element); //помещаем элемент в конец заданного узла
}
}
